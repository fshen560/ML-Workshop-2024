def plot_first_visit(data, plot_by, file_name):
    row = 0
    col = 0

    nrows = 2
    ncols = 3

    data = data.dropna(subset = [plot_by])

    with PdfPages(file_name) as pdf_pages:
        plt.figure()
        fig, axes = plt.subplots(nrows, ncols, figsize = (12, 5))
        fig.tight_layout()

        for i in range(1, len(data.columns)):
            curr = data.iloc[:, i]
            if pd.api.types.is_numeric_dtype(curr):
                if col >= ncols:
                    row += 1
                    col = 0
                if row >= nrows:
                    pdf_pages.savefig(bbox_inches = "tight")
                    plt.figure()
                    fig, axes = plt.subplots(nrows, ncols, figsize = (12, 5))
                    fig.tight_layout()
                    row = 0
                    
                temp = pd.DataFrame({'by': data[plot_by], 'test': curr})
                sns.histplot(temp, ax = axes[row][col], x = curr, bins = 5, 
                             hue = "by", multiple = "dodge")
                col += 1

        pdf_pages.savefig(bbox_inches = "tight")

def plot_time_series(data, plot_by, colors, file_name):
    row = 0
    col = 0

    nrows = 2
    ncols = 3

    data = data.dropna(subset = [plot_by])

    with PdfPages(file_name) as pdf_pages:
        plt.figure()
        fig, axes = plt.subplots(nrows, ncols, figsize = (12, 5))
        fig.tight_layout()

        # change back to (2, 52)
        for i in range(2, 9):
            curr = data.iloc[:, i]
            if pd.api.types.is_numeric_dtype(curr):
                if col >= ncols:
                    row += 1
                    col = 0
                if row >= nrows:
                    plt.legend(title = plot_by, loc = "upper right", labels = data[plot_by].unique())
                    pdf_pages.savefig(bbox_inches = "tight")
                    plt.figure()
                    fig, axes = plt.subplots(nrows, ncols, figsize = (12, 5))

# Function to process data
def process_data(data, features_to_dummy, features_to_scale):
    # One-hot encode categorical features
    ohe = OneHotEncoder(drop='first', sparse_output=False)

    trsf = pd.DataFrame(ohe.fit_transform(data[features_to_dummy]), columns=ohe.get_feature_names_out(features_to_dummy))
    
    # Standardize numerical features
    scaler = StandardScaler()
    scaled = pd.DataFrame(scaler.fit_transform(data.iloc[:, features_to_scale]), columns=data.columns[features_to_scale])

    # Create final DataFrame with target and other features
    data_num = pd.DataFrame({
        'Y': np.where(data['Y'] == 'IS', 0, 1),
        'Sex': np.where(data['Sex'] == 'F', 0, 1)
    }).join(trsf).join(scaled)
    
    return data_num
